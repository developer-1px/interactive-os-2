```tsx
// Grid.tsx — 전체 교체
import React from 'react'

import type { NormalizedData, Plugin, Command } from '../core/types'
import type { BehaviorContext, NodeState } from '../behaviors/types'
import { Aria } from '../components/aria'
import { grid as gridBehavior } from '../behaviors/grid'
import { core } from '../plugins/core'
import { crudCommands } from '../plugins/crud'
import { renameCommands } from '../plugins/rename'
import { dndCommands } from '../plugins/dnd'

interface ColumnDef {
  key: string
  header: string
  field?: string  // entity field를 편집할 때 사용. 없으면 cells[colIndex] 직접 편집
}

interface GridProps {
  data: NormalizedData
  columns: ColumnDef[]
  plugins?: Plugin[]
  onChange?: (data: NormalizedData) => void
  renderCell?: (value: unknown, column: ColumnDef, state: NodeState) => React.ReactNode
  enableEditing?: boolean
  keyMap?: Record<string, (ctx: BehaviorContext) => Command | void>
  'aria-label'?: string
}

const defaultRenderCell = (value: unknown): React.ReactNode => (
  <span>{String(value ?? '')}</span>
)

const editingKeyMap: Record<string, (ctx: BehaviorContext) => Command | void> = {
  'Delete': (ctx) => crudCommands.remove(ctx.focused),
  'F2': (ctx) => renameCommands.startRename(ctx.focused),
  'Alt+ArrowUp': (ctx) => dndCommands.moveUp(ctx.focused),
  'Alt+ArrowDown': (ctx) => dndCommands.moveDown(ctx.focused),
}

export function Grid({
  data,
  columns,
  plugins = [core()],
  onChange,
  renderCell = defaultRenderCell,
  enableEditing = false,
  keyMap,
  'aria-label': ariaLabel,
}: GridProps) {
  const behavior = React.useMemo(() => gridBehavior({ columns: columns.length }), [columns.length])

  const mergedKeyMap = React.useMemo(() => {
    if (!enableEditing && !keyMap) return undefined
    return { ...(enableEditing ? editingKeyMap : {}), ...keyMap }
  }, [enableEditing, keyMap])

  const renderRow = (node: Record<string, unknown>, state: NodeState): React.ReactNode => {
    const cells = (node.data as Record<string, unknown>)?.cells as unknown[] | undefined
    return (
      <div className="grid-row">
        {columns.map((col, i) => (
          <Aria.Cell key={col.key} index={i}>
            {renderCell(cells?.[i], col, state)}
          </Aria.Cell>
        ))}
      </div>
    )
  }

  return (
    <Aria behavior={behavior} data={data} plugins={plugins} onChange={onChange} aria-label={ariaLabel} keyMap={mergedKeyMap}>
      <Aria.Item render={renderRow} />
    </Aria>
  )
}
```

```ts
// src/pages/cms/cmsI18nAdapter.ts
import type { NormalizedData } from '../../interactive-os/core/types'
import { ROOT_ID } from '../../interactive-os/core/types'
import { localeFieldsOf } from './cms-schema'
import { LOCALES } from './cms-types'
import type { LocaleMap } from './cms-types'

export const I18N_COLUMNS = [
  { key: 'key', header: 'Key' },
  ...LOCALES.map(l => ({ key: l, header: l })),
]

/**
 * CMS store → Grid NormalizedData 변환.
 * 각 row의 data에 sourceEntityId, sourceField 메타데이터를 포함하여
 * 역매핑(Grid row → CMS entity) 가능.
 */
export function translatableEntriesToGrid(store: NormalizedData): NormalizedData {
  const rows: string[] = []
  const entities: NormalizedData['entities'] = {}

  for (const [id, entity] of Object.entries(store.entities)) {
    if (id.startsWith('__')) continue
    const d = entity.data as Record<string, unknown> | undefined
    if (!d) continue
    const type = d.type as string | undefined
    if (!type) continue
    for (const field of localeFieldsOf(type)) {
      const value = d[field]
      if (value && typeof value === 'object' && 'ko' in value) {
        const localeMap = value as LocaleMap
        const rowId = `${id}::${field}`
        rows.push(rowId)
        entities[rowId] = {
          id: rowId,
          data: {
            cells: [`${id}.${field}`, ...LOCALES.map(l => localeMap[l] ?? '')],
            sourceEntityId: id,
            sourceField: field,
          },
        }
      }
    }
  }

  return {
    entities,
    relationships: { [ROOT_ID]: rows },
  }
}

/**
 * Grid row에서 CMS entity 역매핑 메타데이터 추출.
 * CmsI18nSheet의 onRenameCommit에서 사용.
 */
export function getRowMetadata(
  gridData: NormalizedData,
  rowId: string,
): { sourceEntityId: string; sourceField: string } | null {
  const row = gridData.entities[rowId]
  if (!row) return null
  const d = row.data as Record<string, unknown>
  const sourceEntityId = d?.sourceEntityId as string | undefined
  const sourceField = d?.sourceField as string | undefined
  if (!sourceEntityId || !sourceField) return null
  return { sourceEntityId, sourceField }
}
```

기존 plain HTML table을 Grid 컴포넌트로 교체. **양방향 싱크 메커니즘:** Grid에 `onRenameCommit` 콜백을 추가. Aria.Editable의 confirm 시점에 이 콜백이 호출되면 CmsI18nSheet에서 어댑터의 metadata를 읽어 CMS engine에 dispatch.

#### 양방향 싱크 설계

```
[Grid rename confirm]
  ↓
Aria.Editable → confirmRename(rowId, cellField, newValue) → Grid 내부 store 업데이트
  ↓
Grid onChange 호출 → CmsI18nSheet에서 rename 감지
  ↓
getRowMetadata(gridData, rowId) → { sourceEntityId, sourceField }
  ↓
colIndex로 locale 결정: LOCALES[colIndex - 1] (col 0은 Key)
  ↓
CMS store에서 원본 LocaleMap 읽기 → 해당 locale만 업데이트
  ↓
engine.dispatch(renameCommands.confirmRename(sourceEntityId, sourceField, updatedLocaleMap))
```

핵심: Grid의 onChange에서 `__rename__` entity 상태 변화를 감지하여 rename commit을 판별. rename이 confirm되면(active: false + 이전에 active: true였으면) 위 흐름 실행.

**실제 접근 (단순):** Grid에 새 prop 추가 없이, CmsI18nSheet가 `onChange`에서 이전 gridData와 diff하여 변경된 셀을 감지하고 CMS engine에 반영한다. Grid 내부 store의 rename은 cells 배열 값을 변경하므로, cells diff로 어떤 row의 어떤 col이 바뀌었는지 알 수 있다.

```tsx
// CmsI18nSheet 내부
const prevGridDataRef = useRef(gridData)

const handleChange = useCallback((newGridData: NormalizedData) => {
  const prev = prevGridDataRef.current
  prevGridDataRef.current = newGridData

  // 변경된 row 찾기 — cells diff
  for (const [rowId, entity] of Object.entries(newGridData.entities)) {
    if (rowId.startsWith('__')) continue
    const prevEntity = prev.entities[rowId]
    if (!prevEntity) continue
    const newCells = (entity.data as Record<string, unknown>)?.cells as string[] | undefined
    const prevCells = (prevEntity.data as Record<string, unknown>)?.cells as string[] | undefined
    if (!newCells || !prevCells) continue

    for (let i = 1; i < newCells.length; i++) { // i=0은 Key, skip
      if (newCells[i] !== prevCells[i]) {
        const meta = getRowMetadata(newGridData, rowId)
        if (!meta) continue
        const locale = LOCALES[i - 1] // col 1→ko, 2→en, 3→ja
        const currentMap = (store.entities[meta.sourceEntityId]?.data as Record<string, unknown>)?.[meta.sourceField] as LocaleMap
        if (!currentMap) continue
        const updatedMap: LocaleMap = { ...currentMap, [locale]: newCells[i]! }
        engine.dispatch(renameCommands.confirmRename(meta.sourceEntityId, meta.sourceField, updatedMap))
      }
    }
  }
}, [store, engine])
```

```tsx
// src/pages/cms/CmsI18nSheet.tsx
import { useMemo, useRef, useCallback } from 'react'
import type { NormalizedData } from '../../interactive-os/core/types'
import type { CommandEngine } from '../../interactive-os/core/createCommandEngine'
import type { NodeState } from '../../interactive-os/behaviors/types'
import { Grid } from '../../interactive-os/ui/Grid'
import { core, GRID_COL_ID } from '../../interactive-os/plugins/core'
import { rename, renameCommands } from '../../interactive-os/plugins/rename'
import { history } from '../../interactive-os/plugins/history'
import { focusRecovery } from '../../interactive-os/plugins/focusRecovery'
import { translatableEntriesToGrid, I18N_COLUMNS, getRowMetadata } from './cmsI18nAdapter'
import { LOCALES } from './cms-types'
import type { LocaleMap } from './cms-types'

const plugins = [core(), rename(), history(), focusRecovery()]

interface CmsI18nSheetProps {
  engine: CommandEngine
  store: NormalizedData
  open: boolean
}

export default function CmsI18nSheet({ engine, store, open }: CmsI18nSheetProps) {
  const gridData = useMemo(() => translatableEntriesToGrid(store), [store])
  const prevGridDataRef = useRef(gridData)
  prevGridDataRef.current = gridData

  const handleChange = useCallback((newGridData: NormalizedData) => {
    const prev = prevGridDataRef.current

    for (const [rowId, entity] of Object.entries(newGridData.entities)) {
      if (rowId.startsWith('__')) continue
      const prevEntity = prev.entities[rowId]
      if (!prevEntity) continue
      const newCells = (entity.data as Record<string, unknown>)?.cells as string[] | undefined
      const prevCells = (prevEntity.data as Record<string, unknown>)?.cells as string[] | undefined
      if (!newCells || !prevCells) continue

      for (let i = 1; i < newCells.length; i++) {
        if (newCells[i] !== prevCells[i]) {
          const meta = getRowMetadata(newGridData, rowId)
          if (!meta) continue
          const locale = LOCALES[i - 1]
          if (!locale) continue
          const currentMap = (store.entities[meta.sourceEntityId]?.data as Record<string, unknown>)?.[meta.sourceField] as LocaleMap
          if (!currentMap) continue
          const updatedMap: LocaleMap = { ...currentMap, [locale]: newCells[i]! }
          engine.dispatch(renameCommands.confirmRename(meta.sourceEntityId, meta.sourceField, updatedMap))
        }
      }
    }
  }, [store, engine])

  if (!open) return null

  const renderCell = (value: unknown, column: typeof I18N_COLUMNS[number], state: NodeState) => {
    const strValue = String(value ?? '')
    const isEmpty = strValue === ''
    const isKeyCol = column.key === 'key'

    if (isKeyCol) {
      return <span className="cms-i18n-cell__key">{strValue}</span>
    }

    return (
      <span className={isEmpty ? 'cms-i18n-cell--empty' : ''}>
        {strValue || '\u00a0'}
      </span>
    )
  }

  return (
    <div className="cms-i18n-sheet">
      <div className="cms-i18n-sheet__header">
        i18n — Translation Sheet
      </div>
      <div className="cms-i18n-sheet__grid">
        <div className="cms-i18n-sheet__col-headers">
          {I18N_COLUMNS.map(col => (
            <div key={col.key} className="cms-i18n-sheet__col-header">{col.header}</div>
          ))}
        </div>
        <Grid
          data={gridData}
          columns={I18N_COLUMNS}
          plugins={plugins}
          enableEditing
          onChange={handleChange}
          renderCell={renderCell}
          aria-label="i18n Translation Sheet"
        />
      </div>
    </div>
  )
}
```

```tsx
// src/__tests__/cms-i18n-grid.integration.test.tsx
import { describe, it, expect } from 'vitest'
import { render } from '@testing-library/react'
import userEvent from '@testing-library/user-event'
import { Grid } from '../interactive-os/ui/Grid'
import { core, GRID_COL_ID } from '../interactive-os/plugins/core'
import { rename, renameCommands } from '../interactive-os/plugins/rename'
import { history } from '../interactive-os/plugins/history'
import { focusRecovery } from '../interactive-os/plugins/focusRecovery'
import { translatableEntriesToGrid, I18N_COLUMNS } from '../pages/cms/cmsI18nAdapter'
import { createStore } from '../interactive-os/core/createStore'
import { ROOT_ID, type NormalizedData } from '../interactive-os/core/types'

function cmsFixture(): NormalizedData {
  return createStore({
    entities: {
      'hero-title': {
        id: 'hero-title',
        data: { type: 'text', value: { ko: '제목', en: 'Title', ja: '' } },
      },
    },
    relationships: { [ROOT_ID]: ['hero-title'] },
  })
}

const plugins = [core(), rename(), history(), focusRecovery()]

function renderI18nGrid(cmsStore: NormalizedData) {
  const gridData = translatableEntriesToGrid(cmsStore)
  return render(
    <Grid
      data={gridData}
      columns={I18N_COLUMNS}
      plugins={plugins}
      enableEditing
      aria-label="i18n"
    />
  )
}

describe('i18n Grid boundary cases', () => {
  it('V7: empty cell (ja) allows F2 rename entry', async () => {
    const user = userEvent.setup()
    const { container } = renderI18nGrid(cmsFixture())

    // Navigate to ja column (col 3)
    const row = container.querySelector('[role="row"]')!
    ;(row as HTMLElement).focus()
    await user.keyboard('{End}') // last col = ja

    await user.keyboard('{F2}')
    const editable = container.querySelector('[contenteditable="true"]')
    expect(editable).not.toBeNull()
  })

  it('V10: first row first col, ArrowUp + ArrowLeft do nothing', async () => {
    const user = userEvent.setup()
    const { container } = renderI18nGrid(cmsFixture())

    const row = container.querySelector('[role="row"]')! as HTMLElement
    row.focus()

    await user.keyboard('{ArrowUp}')
    expect(container.querySelector('[role="row"][tabindex="0"]')).toBe(row)

    await user.keyboard('{ArrowLeft}')
    const focusedCell = row.querySelector('[role="gridcell"][tabindex="0"]')
    expect(focusedCell?.getAttribute('aria-colindex')).toBe('1') // still first col
  })
})
```
