**Goal:** metadata 행동 플래그를 축 옵션으로 흡수하여 11축 → 5축(navigate, select, activate, expand, trap) 통합. 축 조합만으로 pattern 완전 정의.

```ts
import type { Entity, Command } from '../core/types'
import type {
  AriaBehavior,
  BehaviorContext,
  FocusStrategy,
  NodeState,
  SelectionMode,
} from '../behaviors/types'

// v2: KeyMap (기존 v1 호환) + StructuredAxis
export type KeyMap = Record<string, (ctx: BehaviorContext) => Command | void>

export interface AxisConfig {
  focusStrategy: FocusStrategy
  expandable: boolean
  selectionMode: SelectionMode
  activateOnClick: boolean
  followFocus: boolean
  colCount: number
}

export type Axis = KeyMap | { keyMap: KeyMap; config?: Partial<AxisConfig> }

export interface Identity {
  role: string
  childRole?: string
  ariaAttributes: (node: Entity, state: NodeState) => Record<string, string>
}

// v1 호환: PatternConfig도 identity로 사용 가능 (기존 behavior 깨지지 않음)
export type PatternConfig = Omit<AriaBehavior, 'keyMap'>

function isStructuredAxis(axis: Axis): axis is { keyMap: KeyMap; config?: Partial<AxisConfig> } {
  return axis != null && 'keyMap' in axis && typeof (axis as Record<string, unknown>).keyMap === 'object'
}

function extractKeyMap(axis: Axis): KeyMap {
  return isStructuredAxis(axis) ? axis.keyMap : axis
}

function extractConfig(axis: Axis): Partial<AxisConfig> | undefined {
  return isStructuredAxis(axis) ? axis.config : undefined
}

const DEFAULT_FOCUS_STRATEGY: FocusStrategy = { type: 'natural-tab-order', orientation: 'vertical' }

export function composePattern(identity: Identity | PatternConfig, ...axes: Axis[]): AriaBehavior {
  // Collect all keys from all axes
  const allKeys = new Set<string>()
  for (const axis of axes) {
    const km = extractKeyMap(axis)
    for (const key of Object.keys(km)) {
      allKeys.add(key)
    }
  }

  // Build merged keyMap (chain of responsibility)
  const keyMap: KeyMap = {}
  for (const key of allKeys) {
    const handlers = axes.map((axis) => extractKeyMap(axis)[key]).filter(Boolean)
    if (handlers.length === 1) {
      keyMap[key] = handlers[0]
    } else {
      keyMap[key] = (ctx: BehaviorContext): Command | void => {
        for (const handler of handlers) {
          const result = handler(ctx)
          if (result !== undefined) return result
        }
        return undefined
      }
    }
  }

  // Merge config from axes (later overwrites earlier)
  const mergedConfig: Partial<AxisConfig> = {}
  for (const axis of axes) {
    const config = extractConfig(axis)
    if (config) Object.assign(mergedConfig, config)
  }

  // Build AriaBehavior — identity fields + merged config + keyMap
  // PatternConfig (v1) already has focusStrategy etc, so spread them first
  // then overwrite with mergedConfig from axes
  const result: AriaBehavior = {
    ...identity,
    focusStrategy: mergedConfig.focusStrategy
      ?? (identity as PatternConfig).focusStrategy
      ?? DEFAULT_FOCUS_STRATEGY,
    keyMap,
  }

  // Apply merged config fields
  if (mergedConfig.expandable !== undefined) result.expandable = mergedConfig.expandable
  if (mergedConfig.selectionMode !== undefined) result.selectionMode = mergedConfig.selectionMode
  if (mergedConfig.activateOnClick !== undefined) result.activateOnClick = mergedConfig.activateOnClick
  if (mergedConfig.followFocus !== undefined) result.followFocus = mergedConfig.followFocus
  if (mergedConfig.colCount !== undefined) result.colCount = mergedConfig.colCount

  // v1 compat: if identity had these fields and no axis overrode them, keep them
  const ic = identity as PatternConfig
  if (result.expandable === undefined && ic.expandable !== undefined) result.expandable = ic.expandable
  if (result.selectionMode === undefined && ic.selectionMode !== undefined) result.selectionMode = ic.selectionMode
  if (result.activateOnClick === undefined && ic.activateOnClick !== undefined) result.activateOnClick = ic.activateOnClick
  if (result.followFocus === undefined && ic.followFocus !== undefined) result.followFocus = ic.followFocus
  if (result.colCount === undefined && ic.colCount !== undefined) result.colCount = ic.colCount

  return result
}
```

```ts
import type { BehaviorContext } from '../behaviors/types'
import type { AxisConfig, KeyMap } from './composePattern'

interface NavigateOptions {
  orientation?: 'vertical' | 'horizontal' | 'both'
  wrap?: boolean
  grid?: { columns: number }
}

export function navigate(options?: NavigateOptions): { keyMap: KeyMap; config: Partial<AxisConfig> } {
  const opts = options ?? {}
  const wrapOpt = opts.wrap ? { wrap: true } : undefined

  // Grid mode
  if (opts.grid) {
    const keyMap: KeyMap = {
      ArrowDown: (ctx) => ctx.focusNext(),
      ArrowUp: (ctx) => ctx.focusPrev(),
      ArrowRight: (ctx) => ctx.grid?.focusNextCol() ?? ctx.focusNext(),
      ArrowLeft: (ctx) => ctx.grid?.focusPrevCol() ?? ctx.focusPrev(),
      Home: (ctx) => ctx.grid?.focusFirstCol() ?? ctx.focusFirst(),
      End: (ctx) => ctx.grid?.focusLastCol() ?? ctx.focusLast(),
      'Mod+Home': (ctx) => ctx.focusFirst(),
      'Mod+End': (ctx) => ctx.focusLast(),
    }
    return {
      keyMap,
      config: {
        focusStrategy: { type: 'roving-tabindex', orientation: 'vertical' },
        colCount: opts.grid.columns,
      },
    }
  }

  const orientation = opts.orientation ?? 'vertical'
  const keyMap: KeyMap = {}

  if (orientation === 'vertical' || orientation === 'both') {
    keyMap.ArrowDown = (ctx) => ctx.focusNext(wrapOpt)
    keyMap.ArrowUp = (ctx) => ctx.focusPrev(wrapOpt)
  }

  if (orientation === 'horizontal' || orientation === 'both') {
    keyMap.ArrowRight = (ctx) => ctx.focusNext(wrapOpt)
    keyMap.ArrowLeft = (ctx) => ctx.focusPrev(wrapOpt)
  }

  if (orientation !== 'both') {
    keyMap.Home = (ctx) => ctx.focusFirst()
    keyMap.End = (ctx) => ctx.focusLast()
  }

  return {
    keyMap,
    config: {
      focusStrategy: { type: 'roving-tabindex', orientation },
    },
  }
}
```

import type { AxisConfig, KeyMap } from './composePattern'
import type { SelectionMode } from '../behaviors/types'

interface SelectOptions {
  mode?: SelectionMode
  extended?: boolean
}

export function select(options?: SelectOptions): { keyMap: KeyMap; config: Partial<AxisConfig> } {
  const mode = options?.mode ?? 'multiple'
  const extended = options?.extended && mode === 'multiple'

  const keyMap: KeyMap = {
    Space: (ctx) => ctx.toggleSelect(),
  }

  if (extended) {
    keyMap['Shift+ArrowDown'] = (ctx) => ctx.extendSelection('next')
    keyMap['Shift+ArrowUp'] = (ctx) => ctx.extendSelection('prev')
    keyMap['Shift+Home'] = (ctx) => ctx.extendSelection('first')
    keyMap['Shift+End'] = (ctx) => ctx.extendSelection('last')
  }

  return { keyMap, config: { selectionMode: mode } }
}

import type { AxisConfig, KeyMap } from './composePattern'

interface ActivateOptions {
  onClick?: boolean
  followFocus?: boolean
  toggleExpand?: boolean
}

export function activate(options?: ActivateOptions): { keyMap: KeyMap; config: Partial<AxisConfig> } {
  const keyMap: KeyMap = {
    Enter: (ctx) => ctx.activate(),
    Space: (ctx) => ctx.activate(),
  }

  const config: Partial<AxisConfig> = {}
  if (options?.onClick) config.activateOnClick = true
  if (options?.followFocus) config.followFocus = true
  if (options?.toggleExpand) config.expandable = true

  return { keyMap, config }
}

```ts
import type { AxisConfig, KeyMap } from './composePattern'
import { createBatchCommand, ROOT_ID } from '../core/types'
import { spatialCommands, SPATIAL_PARENT_ID } from '../plugins/spatial'
import { focusCommands } from '../plugins/core'
import { renameCommands } from '../plugins/rename'

interface ExpandOptions {
  mode?: 'arrow' | 'enter-esc'
}

export function expand(options?: ExpandOptions): { keyMap: KeyMap; config: Partial<AxisConfig> } {
  const mode = options?.mode ?? 'arrow'

  if (mode === 'enter-esc') {
    const keyMap: KeyMap = {
      Enter: (ctx) => {
        const children = ctx.getChildren(ctx.focused)
        if (children.length > 0) {
          return createBatchCommand([
            spatialCommands.enterChild(ctx.focused),
            focusCommands.setFocus(children[0]),
          ])
        }
        return renameCommands.startRename(ctx.focused)
      },
      Escape: (ctx) => {
        const spatialParent = ctx.getEntity(SPATIAL_PARENT_ID)
        const parentId = spatialParent?.parentId as string | undefined
        if (!parentId || parentId === ROOT_ID) return undefined
        return createBatchCommand([
          spatialCommands.exitToParent(),
          focusCommands.setFocus(parentId),
        ])
      },
    }
    return { keyMap, config: { expandable: true } }
  }

  // arrow mode
  const keyMap: KeyMap = {
    ArrowRight: (ctx) => (ctx.isExpanded ? ctx.focusChild() : ctx.expand()),
    ArrowLeft: (ctx) => (ctx.isExpanded ? ctx.collapse() : ctx.focusParent()),
  }
  return { keyMap, config: { expandable: true } }
}
```

## Review Errata (리뷰 반영 수정사항)

implementer는 아래 사항을 반드시 반영:

### E1. navigate() grid 모드 orientation

navigate factory의 grid 모드에서 `orientation: 'vertical'` → `orientation: 'both'`로 변경. v1 grid behavior의 `focusStrategy: { type: 'roving-tabindex', orientation: 'both' }`와 일치해야 함.

### E2. combobox.ts는 navigate() 사용 금지

combobox의 focusStrategy는 `aria-activedescendant`이므로 navigate() (roving-tabindex 기본) 사용 시 parity 깨짐. combobox는 `navV`를 인라인 KeyMap으로 유지하고, focusStrategy를 identity에 직접 지정:

```ts
composePattern(
  { role: 'combobox', childRole: 'option',
    focusStrategy: { type: 'aria-activedescendant', orientation: 'vertical' },
    ariaAttributes: ... },
  popupToggle,      // 인라인 KeyMap
  { ArrowDown: (ctx) => ctx.focusNext(), ArrowUp: (ctx) => ctx.focusPrev(), ... },  // 인라인 navV KeyMap
)
```

composePattern의 v1 호환 코드가 identity의 focusStrategy를 유지하므로, 축에서 focusStrategy를 덮어쓰지 않으면 identity 값이 사용됨.

### E3. spatial.ts에 activate({ onClick: true }) 추가

v1 spatial에는 `activateOnClick: true`가 있음. v2에서도 동일하게:

```ts
export const spatial = composePattern(
  { role: 'group', childRole: 'group', ariaAttributes: ... },
  select(),
  activate({ onClick: true }),
  expand({ mode: 'enter-esc' }),
  spatialNav,  // 인라인 KeyMap 유지
)
```

### E4. kanban.ts selectionMode 주의

v1 kanban은 `selectionMode`를 명시하지 않음. `select()`를 사용하면 `selectionMode: 'multiple'`이 추가됨. 이것이 의도된 변경인지 확인 필요. 안전한 방법: selectToggle을 인라인 KeyMap으로 유지.

```ts
const selectToggleKeyMap: KeyMap = { Space: (ctx) => ctx.toggleSelect() }
```

### E5. activate-v2 테스트 파일 rename

Task 7 Step 1의 테스트 파일 mv 명령이 no-op. 파일 내부 import를 `../activate-v2` → `../activate`로 변경만 하면 됨.
